#form.py

class ClienteForm(forms.Form):
 
    SEXO = (
        (1, "Masculino"),
        (2, "Feminino"),
    )
    sexo = forms.ChoicerField(choices=SEXO, max_length=1)


Assim fica blz, mas as vezes queremos torná-lo mais dinâmico e buscar esses dados do próprio banco e agora que entra a dica.

Vamos supor que temos as classes Cliente e Sexo no models.py

# models.py

class Sexo(models.Model):
    descricao = model.CharField(max_length=10)
 
class Cliente(models.Model):
    nome = models.CharField(max_length=100)
    sexo = models.ForeignKey(Sexo)


# forms.py

from app.models import Sexo
class ClienteForm(forms.Form):
    SEXO = Sexo.objects.values_list('id', 'descricao')
    sexo = forms.ChoiceField(choices=SEXO, max_length=1)
e se quiser que apareça como primeira opção no combo aqueles tracinhos (‘——–‘), basta fazer assim.

# forms.py

from app.models import Sexo
class ClienteForm(forms.Form):
    SEXO = ['', '-------'] + [(s.id, s.descricao) for s in Sexo.objects.all().order_by('descricao')]
    sexo = forms.ChoiceField(choices=SEXO, max_length=1)
